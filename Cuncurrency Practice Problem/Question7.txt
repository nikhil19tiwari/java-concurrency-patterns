package nikhil.co.in.siem.config;

/**
Problem 7: Visibility Control Flag
Create a class Worker:
One thread updates a flag (stop)
Another thread loops until flag becomes true
Requirements:
Ensure worker thread stops reliably
Avoid busy-wait visibility issues
*/

class Worker{
	private volatile boolean flag = true;
	public void doWork() {
		int i =0;
	while(flag) {
	System.out.println(i);
	i++;
	}

	}
	public void stop() {
		flag = false;
	}
}
public class Test {
    public static void main(String[] args) throws InterruptedException {
    	Worker w = new Worker();
    	
    Thread t1 = new Thread(()-> {w.doWork(); });
    t1.start();
    Thread.sleep(10);
    w.stop();
   
    }
}
